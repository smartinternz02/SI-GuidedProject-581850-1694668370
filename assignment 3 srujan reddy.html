def find_oulier_percentage(data:pd.DataFrame,column:str) -> None:
    q1,q3 = np.percentile(data[column].dropna(),[25,75])
    IQR = q3-q1
    min_ = q1 - IQR*1.5
    max_ = q3 + IQR*1.5
    len_ = data[(data[column] > max_ ) | (data[column] < min_)].shape[0]
    print(column, format(len_*100/data.shape[0],'.2f'))
find_oulier_percentage(data=data,column='Age')

def normalize(data:pd.DataFrame,column:str):
    mean = data[column].mean()
    std = data[column].std()
    data[column] = data[column].apply(lambda x : (x-mean)/std)
